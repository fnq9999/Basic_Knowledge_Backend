```cgo
### 堆排序
class heap{ //// min heap in head
private:
    int cap=0;
    int a[30];
    int sz=0;
public:
    heap(){memset(a,0,sizeof(a));}
    void push(int x){
        a[++sz]=x;
        int ind=sz;
        while(ind!=1){
            if(a[ind]<a[ind/2]){
                swap(a[ind/2],a[ind]);
                ind/=2;
            }
            else{
                break;
            }
        }
    }
    int pop(){
        int tem=a[1];
        a[1]=a[sz--];
        while(ind*2<=sz){
            int son=ind;
            if(a[ind]>a[ind*2]){
                son=ind*2;
            }
            if(ind*2+1<=sz and a[ind*2+1]<a[ind*2] and a[ind*2+1]<a[ind]){
                son=ind*2+1;
            }
            if(!(son-ind))break;
            swap(a[ind],a[son]);
            ind=son;
        }
        return tem;
    }
};
```

### leetcode 327 区间和的个数<br>

		问你区间中，有多少个区间和满足 lower<=sum<=upper
		解：
		upper>=sum[j]-sum[i]>=lower ---->  sum[i]+upper>=sum[j]>=sum[i]+lower;
		然后倒序(n-->0)离散化 查找sum[j]满足落在上所述区间里的个数（树状数组）。
```c++
class Solution {
public:
    #define ll long long
    vector<ll>vec;
    #define pb push_back
    vector<ll>c;
    #define lowbit(x) (x&(-x))
    void update(ll x){
        for(;x<=c.size()-1;x+=lowbit(x)){
            c[x]+=1;
        }
    }
    int ask(ll x){
        ll res=0;
        for(;x;x-=lowbit(x)){
            res+=c[x];
        }
        return res;
    }
    int find(ll x){//// 找到小于等于X第一个
        int l=0,r=vec.size()-1;
        while(l<r){
            ll mid=(l+r+1)>>1;
            if(vec[mid]<=x){
                l=mid;
            }else{
                r=mid-1;
            }
        }
        return l;
    }
    int countRangeSum(vector<int>& nums, int lower, int upper) {
        ll sum=0;
        vec.pb(sum);
        int n=nums.size();
        vector<ll>s(n+5,0);
        s[0]=0;
        for(int i=0;i<n;++i){
            s[i+1]=nums[i]+s[i];
            vec.pb(s[i+1]+lower-1);
            vec.pb(s[i+1]+upper);
            vec.pb(s[i+1]);
        }
        vec.pb(-0x3f3f3f3f3f3f3f3f);
        sort(vec.begin(),vec.end());
        vec.resize(unique(vec.begin(),vec.end())-vec.begin());///离散化去重
        c=vector<ll>(vec.size()+2,0);//// 树状数组大小
        ll ans=0;
        for(int i=n;i>=0;--i){
            ans+=ask(find(s[i]+upper))-ask(find(s[i]+lower-1));
            int ind3=find(s[i]);
            update(ind3);
        }
        return ans;
    }
};
```



- leetcode  164最大间距
  -给你一个无序数组，问你排完序之后，相邻两个元素的差值的最大值是多少？O(N)时间。
```c++
struct bkt
{
	bool used = false;
	int mx = INT_MIN;
	int mi = INT_MAX;
};
class Solution {
public:
	int maximumGap(vector<int>& nums) {
		if (nums.size() < 2)
			return 0;
		int res = 0;
		//nums里面最大最小值
        #define inf 0x3f3f3f3f
        int maxnum=-inf;
        int minnum=inf;
        for(int i=0;i<nums.size();++i){
            maxnum=max(maxnum,nums[i]);
            minnum=min(minnum,nums[i]);
        }
		//桶容量大小
        int bktsz=max(1,int((maxnum-minnum)/(nums.size()-1)));
		//对于bucketSize=0情况，桶大小为0，每个桶里一个元素都存不满，是因为nums中存在大量重复元素
		//桶个数
		int bktnum=(maxnum-minnum)/(bktsz)+1;
		//每个桶里只保存桶中最大元素和最小元素
		vector<bkt> bkts(bktnum);
		for (auto& v : nums)
		{
			int pos = (v - minnum) / bktsz;
			bkts[pos].used = true;
			bkts[pos].mx = max(bkts[pos].mx, v);
			bkts[pos].mi = min(bkts[pos].mi, v);
		}
		int pre = minnum;
		//遍历每个桶，最大间距出现在两个桶里的元素差值里
		for (int i = 0; i < bkts.size(); i++)
		{
			if (!bkts[i].used)
				continue;
			//当前桶里最小值和前一个桶里最大值的差值可能是nums的最大间距
			res = max(res, bkts[i].mi - pre);
			//前一个桶的最大值
			pre = bkts[i].mx;
		}
		return res;
	}
};
```

### leetcode 57 插入区间<br>
		 题意：向一个已经排好序区间数列插入一个新的区间，使得返回一个不重叠的区间数组。
		 解法：一次遍历，第一次遇到原数列中左端点大于现在端点的时候，先考虑插入new interval,
		 然后正常考虑别的区间。（最后也要考虑老的区间中，没有一个的左端点大于new interval）
```c++
class Solution {
public:
    vector<vector<int>> insert(vector<vector<int>>& intervals, vector<int>& newInterval) {
        if(intervals.size()==0){
            return {newInterval};
        }
        bool yes=0;
        vector<vector<int>>ans;
        int n=intervals.size();
        vector<int>pre;
        for(int i=0;i<n;i++){
            if(!yes and newInterval[0]<=intervals[i][0]){
                if(pre.size()){
                    if(newInterval[0]>pre[1]){
                        ans.push_back(pre);
                        pre=newInterval;
                    }else{
                        pre[1]=max(pre[1],newInterval[1]);
                    }
                }else{
                    pre=newInterval;
                }
                yes=1;
            }
                if(pre.size()){
                    if(intervals[i][0]>pre[1]){
                        ans.push_back(pre);
                        pre=intervals[i];
                    }else{
                        pre[1]=max(pre[1],intervals[i][1]);
                    }
                }else{
                    pre=intervals[i];
                }
        }
        if(!yes){
            if(newInterval[0]>pre[1]){
                ans.push_back(pre);
                pre=newInterval;
            }else{
                pre[1]=max(pre[1],newInterval[1]);
            }
        }
        if(pre.size()){
            ans.push_back(pre);
        }
        
        return ans;
    }
};
```
