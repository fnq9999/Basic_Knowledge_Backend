### leetcode 162寻找峰值<br>

```c++
class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        if(nums.size()==1)return 0;
        int l=-1,r=nums.size();
        while(l+1<r){
            int lmid=(l+r)>>1;
            int rmid=(lmid+r)>>1;
            int left=lmid<0?0xcfcfcfcf:nums[lmid];
            int right=rmid<nums.size()?nums[rmid]:0x3f3f3f3f;
            if(left<=right){
                l=lmid;
            }else{
                r=rmid;
            }
        }
        return l;
    }
};
```
