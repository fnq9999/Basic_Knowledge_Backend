### leetcode 146 LRU缓存机制<br>
        让你实现一个LRU缓存数据结构
```c++

struct Node{
    Node* nxt=NULL,* pre=NULL;
    int key,val;
    Node(){}
    Node(int _key,int _val){key=_key,val=_val;}
};
class LRUCache {
public:
    Node* head,* tail;
    int sz=0,cap;
    unordered_map<int,Node*>cache;
    LRUCache(int capacity) {
        cap=capacity;
        head=new Node();
        tail=new Node();
        head->nxt=tail;
        tail->pre=head;
    }  
    void del(Node * node){
        node->pre->nxt=node->nxt;
        node->nxt->pre=node->pre;
    }
    void add_to_head(Node* node){
        node->nxt=head->nxt;
        head->nxt->pre=node;
        head->nxt=node;
        node->pre=head;
    }
    void rm_to_head(Node *node){
        del(node);
        add_to_head(node);
    }
    int get(int key) {
        if(cache.count(key)==0)return -1;
        Node* node =cache[key];
        rm_to_head(node);
        return node->val;
    }
    void put(int key, int value) {
        if(cache.count(key)){
            Node* tem=cache[key];
            tem->val=value;
            rm_to_head(tem);
        }else{
            Node* tem=new Node(key,value);
            add_to_head(tem);
            cache[key]=tem;
            ++sz;
            if(sz>cap){
                --sz;
                Node* tem=tail->pre;
                del(tem);
                cache.erase(tem->key);
                delete tem;
            }
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        if(l1==l2 and l1==NULL)return NULL;
        ListNode* ans=new ListNode(0);
        ListNode* p=ans;
        while(l1!=NULL or l2!=NULL){
            if(l1!=NULL){
                ans->val+=l1->val;
                l1=l1->next;
            }
            if(l2!=NULL){
                ans->val+=l2->val;
                l2=l2->next;
            }
            if(l1!=NULL or l2!=NULL or (ans->val)>=10){
                ans->next=new ListNode((ans->val)/10);
            }
            (ans->val)%=10;
            ans=ans->next;
        }
        return p;
    }
};
```

```c++
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* cur=head,* pre=head;
        n+=1;
        while(n-- and pre!=NULL){
            pre=pre->next;
        }
        if(n!=-1){
            return head->next;
        }
        while(pre!=NULL){
            pre=pre->next;
            cur=cur->next;
        }
        cur->next=cur->next->next;
        return head;
    }
};
```
