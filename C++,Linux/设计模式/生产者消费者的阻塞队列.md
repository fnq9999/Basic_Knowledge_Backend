Block_Queue
=====================

&emsp;&emsp;该问题很基础，最基本的模型只需要三个功能
- size() 
- push()
    - 满了的话就广播并返回
    - 放入成功也广播
- pop()
    - 如果队列里面没有任务，就使用 while if  wait来等待别人的唤醒
```c++
<template T>
class blq{
private:
    cond m_cond;
    locker mutex;
    T *a;
    int cap=0;
    int sz=0,l=0,r=-1;/////  l=0,r=-1是一个经典玩法，这样l就代表头，r在push时为a[++r]=new object;,在pop时，pop(a[l--]);

public:////其实l=0,r=0也行，只要用sz保护住就行
    blq(int capcity){cap=capcity;}
    ~blq(){delete []a;}
    int size(){
        int res;
        mutex.lock();////把变量的定义放到外面去了
        res=sz;
        mutex.unlock();
        return res;
    }
    bool push(T &tem){
        mutex.lock();
        if(sz==cap){
            m_cond.broadcast();/// 满了广播
            mutex.unlock();
            return false;
        }
        ++r;r%=cap;++sz;
        a[r]=tem;
        m_cond.broacast();////  刚放进去也广播
        mutex.unlock();
        return true;
    }
    bool pop(T &tem){
        mutex.lock();
        while(0==sz){
            if(!cond.wait(mutex.get()){/////  while if wait结构 比较安全
                mutex.unlock();
                return false;
            }
        }
        tem=a[l];
        ++l;l%=cap;--sz;
        mutex.unlock();
        return true;
    }
};

```
